In [4]:
squares = [1, 4, 9, 16, 25]
plt.plot(squares, linewidth=5)

# Set chart title and label axes.
plt.title("Square Numbers", fontsize=24)
plt.xlabel("Value", fontsize=14)
plt.ylabel("Square of Value", fontsize=14)

# Set size of tick labels.
plt.tick_params(axis='both', labelsize=14)

plt.show()

Note that plots naturally assume X-coordinates start at 0. If the x is not zero, make two separate lists - one for x and one for y - to pass to the plt.plot() function.¶

## define x and y range
x_values = list(range(1,5000))
y_values = [x**3 for x in x_values]

## convert to scatter plot & define size
plt.scatter(x_values, y_values, edgecolor ='none', s=10)

# Set the range for each axis.
plt.axis([1, 5000, 0, 100000000000])

## set chart title, label axis, tickmarks
plt.title("Cubed Numbers", fontsize=15)
plt.xlabel("Number", fontsize=10)
plt.ylabel("Cube of Number", fontsize=10)
plt.tick_params(axis='both', which='major', labelsize=10)

plt.show()

## define x and y range
x_values = list(range(1,5000))
y_values = [x**3 for x in x_values]

## convert to scatter plot, define size, remove edge color, apply colormap
plt.scatter(x_values, y_values, edgecolor ='none', s=10, c=y_values, cmap=plt.cm.Blues)

# Set the range for each axis.
plt.axis([1, 5000, 0, 100000000000])

## set chart title, label axis, tickmarks
plt.title("Cubed Numbers", fontsize=15)
plt.xlabel("Number", fontsize=10)
plt.ylabel("Cube of Number", fontsize=10)
plt.tick_params(axis='both', which='major', labelsize=10)

plt.show()
